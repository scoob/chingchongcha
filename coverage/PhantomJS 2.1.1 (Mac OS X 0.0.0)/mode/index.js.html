(function () { 'use strict';
 
/**
* Module: mode
*
* Controls for managing the game mode
*/
 
var MODULE_NAME = module.exports = 'mode';
 
angular.module(MODULE_NAME, [])
/**
 * @ngdoc service
 * @name dataservices.service:ModeService
 * @param $sessionStorage {object} local storage to access token
 * @description
 * Service to control the game mode
 */
// @ngInject
.service('ModeService', function ModeService($sessionStorage) {
  var service = this;
  // initialise the mode on initial load
  service.init = function () {
    if (!$sessionStorage.mode) {
      $sessionStorage.mode = 0; // 0 = human, 1 = computer
    }
  };
  service.setMode = function (mode) {
    return $sessionStorage.mode = mode;
  };
  service.getMode = function () {
    return $sessionStorage.mode;
  };
})
/**
 * @ngdoc controller
 * @name mode.controller:ModeController
 * @param ModeService {object} session storage to access mode
 * @description
 * Allow mode to be set and retreaved
 */
// @ngInject
.controller('ModeController', function ModeController(ModeService) {
  var controller = this;
  ModeService.init();
  controller.setMode = function (mode) {
    return ModeService.setMode(mode);
  };
  controller.getMode = function () {
    return ModeService.getMode();
  };
})
/**
 * @ngdoc directive
 * @name mode.directive:modeButton
 * @restrict A
 * @description
 * Allow user to set the game mode and update the session storage
 */
// @ngInject
.directive('modeButton', function () {
  return {
    restrict: 'A',
    controller: 'ModeController',
    scope: true,
    link: function link($scope, $element, attr, controller) {
      $scope.mode = controller.getMode() ? true : false;
      $scope.$watch('mode', function (mode) {
        controller.setMode(mode ? true : false);
      });
    }
  };
})
/**
 * @ngdoc directive
 * @name mode.directive:modeState
 * @restrict A
 * @description
 * Access the state of the game mode
 */
// @ngInject
.directive('modeState', function () {
  return {
    restrict: 'A',
    controller: 'ModeController',
    scope: true,
    link: function link($scope, $element, attr, controller) {
      $scope.$watch(controller.getMode, function (mode) {
        $scope.mode = mode;
      });
    }
  };
});

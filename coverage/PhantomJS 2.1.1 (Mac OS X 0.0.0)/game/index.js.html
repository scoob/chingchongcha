/**
 * @ngdoc controller
 * @name game.controller:GameController

 * @param ScoreboardService {object} session storage to access scores
 * @description
 * Initialise the scoreboard and allow the scoreboard to get all scores

// @ngInject
.controller('GameController', function GameController(GameService, ScoreboardService) {
  var controller = this;
  // Useful maps for game control
  var hands = ['rock', 'paper', 'scissors'];
  var winningMap = {
    rock: 'scissors',
    paper: 'rock',
    scissors: 'paper'
  };
  var playerMap = {
    player1: 'Player 1',
    player2: 'Player 2',
    tie: 'Tie'
  };
  // initialise player choices
  var player1 = null;
  var player2 = null;
  var isPlaying = false;
  GameService.init();
  // play the game and get the winner
  controller.play = function (hand) {
    player1 = GameService.getPlayer1(hand, hands);
    player2 = GameService.getPlayer2(hand, hands);
    var winner = GameService.getWinner(winningMap, player1, player2);
    GameService.addResult(playerMap[winner], player1, player2);
    return winner;
  };
 
  controller.updateScore = function (winner) {
    return ScoreboardService.updateScore(winner);
  };
  controller.getLatestResult = function () {
    return GameService.getLatestResult();
  };
  controller.getPlayers = function () {
    return GameService.getPlayers();
  };
  controller.togglePlayingState = function () {
    return isPlaying = !isPlaying;
  };
  controller.getPlayingState = function () {
    return isPlaying;
  };
})
/**
 * @ngdoc directive
 * @name game.directive:playButton
 * @param $timeout {function} force a delay
 * @restrict A
 * @description
 * Register a player's click on a selected hand button
 */
// @ngInject
.directive('playButton', function ($timeout) {
  return {
    restrict: 'A',
    controller: 'GameController',
    scope: true,
    link: function link($scope, $element, attr, controller) {
      $element.on('click', function () {
        $element.addClass('active'); // highlight the chosen button
        // let player know what the computer is doing
        var loadingCircle = angular.element(document.getElementById('loading-circle')).addClass('active');
        controller.togglePlayingState();
        // make out like the computer is thinking
        $timeout(function () {
          var winner = controller.play(attr.type);
          controller.updateScore(winner);
          controller.togglePlayingState();
          loadingCircle.removeClass('active');
        }, 3000);
        // keep chosen button highlighted a little longer
        $timeout(function () {
          $element.removeClass('active');
        }, 4000);
      });
    }
  };
})
